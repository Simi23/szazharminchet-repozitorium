# 1-create-ou.ps1
# Creates one or more Organizational Units under the domain root.
# Usage:
#   .\1-create-ou.ps1 "HR" "IT" "Finance"

param (
   [Parameter(Mandatory = $true)]
   [string[]]$OuNames
)

# Import the Active Directory module
Import-Module ActiveDirectory

function Get-DomainRootDn {
   # Get the distinguished name of the current AD domain
   return (Get-ADDomain).DistinguishedName
}

function Test-OUExists {
   param (
       [Parameter(Mandatory = $true)]
       [string]$OuName,

       [Parameter(Mandatory = $true)]
       [string]$BaseDn
   )

   # Check whether the OU already exists directly under the domain root
   $existingOu = Get-ADOrganizationalUnit `
       -Filter "Name -eq '$OuName'" `
       -SearchBase $BaseDn `
       -SearchScope OneLevel `
       -ErrorAction SilentlyContinue

   return $null -ne $existingOu
}

function New-DomainOU {
   param (
       [Parameter(Mandatory = $true)]
       [string]$OuName,

       [Parameter(Mandatory = $true)]
       [string]$BaseDn
   )

   # Create the OU only if it does not already exist
   if (Test-OUExists -OuName $OuName -BaseDn $BaseDn) {
       Write-Host "OU already exists: $OuName"
   }
   else {
       New-ADOrganizationalUnit `
           -Name $OuName `
           -Path $BaseDn `
           -ProtectedFromAccidentalDeletion $true

       Write-Host "OU created: $OuName"
   }
}

# Main script
$domainRootDn = Get-DomainRootDn

foreach ($ouName in $OuNames) {
   New-DomainOU -OuName $ouName -BaseDn $domainRootDn
}

 

Delete OU

# 2-delete-ou.ps1
# Deletes one or more Organizational Units from the domain root.
# Usage:
#   .\2-delete-ou.ps1 "HR" "IT" "Finance"

param (
   [Parameter(Mandatory = $true)]
   [string[]]$OuNames
)

# Import the Active Directory module
Import-Module ActiveDirectory

function Get-DomainRootDn {
   # Get the distinguished name of the current AD domain
   return (Get-ADDomain).DistinguishedName
}

function Get-DomainOU {
   param (
       [Parameter(Mandatory = $true)]
       [string]$OuName,

       [Parameter(Mandatory = $true)]
       [string]$BaseDn
   )

   # Find the OU directly under the domain root
   return Get-ADOrganizationalUnit `
       -Filter "Name -eq '$OuName'" `
       -SearchBase $BaseDn `
       -SearchScope OneLevel `
       -ErrorAction SilentlyContinue
}

function Remove-DomainOU {
   param (
       [Parameter(Mandatory = $true)]
       [string]$OuName,

       [Parameter(Mandatory = $true)]
       [string]$BaseDn
   )

   # Check whether the OU exists before trying to delete it
   $ou = Get-DomainOU -OuName $OuName -BaseDn $BaseDn

   if ($null -eq $ou) {
       Write-Host "OU does not exist: $OuName"
       return
   }

   # Disable accidental deletion protection if enabled
   if ($ou.ProtectedFromAccidentalDeletion) {
       Set-ADOrganizationalUnit `
           -Identity $ou.DistinguishedName `
           -ProtectedFromAccidentalDeletion $false

       Write-Host "Accidental deletion protection disabled: $OuName"
   }

   # Delete the OU
   Remove-ADOrganizationalUnit `
       -Identity $ou.DistinguishedName `
       -Confirm:$false

   Write-Host "OU deleted: $OuName"
}

# Main script
$domainRootDn = Get-DomainRootDn

foreach ($ouName in $OuNames) {
   Remove-DomainOU -OuName $ouName -BaseDn $domainRootDn
}

# 3-create-user.ps1
# Creates an Active Directory user.
# Usage:
#   .\3-create-user.ps1 "jsmith"
#   .\3-create-user.ps1 "jsmith" "Sales"

param (
    [Parameter(Mandatory = $true, Position = 0)]
    [string]$Username,

    [Parameter(Mandatory = $false, Position = 1)]
    [string]$OuName
)

# Import the Active Directory module
Import-Module ActiveDirectory

function Get-DomainRootDn {
    # Get the distinguished name of the current AD domain
    return (Get-ADDomain).DistinguishedName
}

function Get-DomainDnsName {
    # Get the DNS name of the current AD domain
    return (Get-ADDomain).DNSRoot
}

function Get-TargetPath {
    param (
        [string]$OuName,
        [string]$DomainRootDn
    )

    # If no OU is specified, create the user under the domain root
    if ([string]::IsNullOrWhiteSpace($OuName)) {
        return $DomainRootDn
    }

    # Search for the specified OU directly under the domain root
    $ou = Get-ADOrganizationalUnit `
        -Filter "Name -eq '$OuName'" `
        -SearchBase $DomainRootDn `
        -SearchScope OneLevel `
        -ErrorAction SilentlyContinue

    if ($null -eq $ou) {
        throw "OU does not exist under the domain root: $OuName"
    }

    return $ou.DistinguishedName
}

function Test-ADUserExists {
    param (
        [Parameter(Mandatory = $true)]
        [string]$Username
    )

    # Check whether the user already exists
    $user = Get-ADUser `
        -Filter "SamAccountName -eq '$Username'" `
        -ErrorAction SilentlyContinue

    return $null -ne $user
}

function New-DomainUser {
    param (
        [Parameter(Mandatory = $true)]
        [string]$Username,

        [Parameter(Mandatory = $true)]
        [string]$TargetPath,

        [Parameter(Mandatory = $true)]
        [string]$DomainDnsName
    )

    # Do nothing if the user already exists
    if (Test-ADUserExists -Username $Username) {
        Write-Host "User already exists: $Username"
        return
    }

    # Default initial password
    # The user must change it at first logon
    $initialPassword = ConvertTo-SecureString "P@ssw0rd123!" -AsPlainText -Force

    # Create the user
    New-ADUser `
        -Name $Username `
        -SamAccountName $Username `
        -UserPrincipalName "$Username@$DomainDnsName" `
        -Path $TargetPath `
        -AccountPassword $initialPassword `
        -Enabled $true `
        -ChangePasswordAtLogon $true

    Write-Host "User created: $Username"
}

# Main script
$domainRootDn = Get-DomainRootDn
$domainDnsName = Get-DomainDnsName
$targetPath = Get-TargetPath -OuName $OuName -DomainRootDn $domainRootDn

New-DomainUser `
    -Username $Username `
    -TargetPath $targetPath `
    -DomainDnsName $domainDnsName

# 4-delete-user.ps1
# Deletes an Active Directory user.
# Usage:
#   .\4-delete-user.ps1 "jsmith"

param (
    [Parameter(Mandatory = $true, Position = 0)]
    [string]$Username
)

# Import the Active Directory module
Import-Module ActiveDirectory

function Get-DomainUser {
    param (
        [Parameter(Mandatory = $true)]
        [string]$Username
    )

    # Find the user by username
    return Get-ADUser `
        -Filter "SamAccountName -eq '$Username'" `
        -ErrorAction SilentlyContinue
}

function Remove-DomainUser {
    param (
        [Parameter(Mandatory = $true)]
        [string]$Username
    )

    # Check whether the user exists before trying to delete it
    $user = Get-DomainUser -Username $Username

    if ($null -eq $user) {
        Write-Host "User does not exist: $Username"
        return
    }

    # Delete the user
    Remove-ADUser `
        -Identity $user.DistinguishedName `
        -Confirm:$false

    Write-Host "User deleted: $Username"
}

# Main script
Remove-DomainUser -Username $Username
